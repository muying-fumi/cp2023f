請定義3個function: f1(n), f2(n1, n2), main()。

f1(n):
Input: n is an Integer.
Output: If the n is not in {1, 2, 3, 4,…, 30}, then output "Error".

f2(n1, n2):
Input：n1, n2 are Integer.
Output：
(1) If n1 not in {1, 2, 3}, then output "Error".
(2) If n2 not in {1, 3, 5, 7, 9, …,29}, then output "Error".
(3) n1 denotes 1 (right Triangle diagram), 2 (left Triangle diagram), 3 (Diamond diagram)}.
(4) n2 denotes the height of the diagram.

main():
User can input repeatedly until 0 to exit the program.
(可一直輸入直到輸入 0 結束；輸出 Error 結束)
Input 1, then input n to run f1(n);
Input 2, then input n1, n2 to run f2(n1, n2);
Other input, then output "Error" to exit the program.

在main()裡依據使用者輸入的數字來呼叫f1(n)或f2(n1, n2)，要能重複輸入。

            #include <stdio.h>

            void f1(int n){
              int i,j;
              if (n<1 || n>30){
                printf("Error\n");
                return;
              }
              for (i=1;i<=n;i++){
                for (j=1;j<=i;j++)
                  printf("*"); //每行有i個星號
                for (j=n-i+1;j>=1;j--){
                  if (j<=9)
                    printf("%d",j);
                  else
                    printf("%c",(j-10)+'a');//以小寫英文印出
                }
                printf("\n"); //印完一行要換行
              }
            }

            void f2(int n1, int n2){
              int i,j;
              if (n1<1 || n1>3){
                printf("Error\n");
                return;
              }
              if (n2<1 || n2>29 || n2%2==0){ //只能1~29的奇數
                printf("Error\n");
                return;
              }
              //n1=1: right triangle
              if (n1==1){
                for (i=1;i<=n2/2;i++){ //上半
                  for (j=1;j<=i;j++)
                    printf("*");
                  printf("\n");
                }
                for (i=n2/2+1;i>=1;i--){ //下半
                  for (j=1;j<=i;j++)
                    printf("*");
                  printf("\n");
                }
              }
              else if (n1==2){ //n1=2: left triangle
                for (i=1;i<=n2/2;i++){ //上半
                  for (j=1;j<=n2/2+1-i;j++)
                    printf("#");
                  for (j=1;j<=i;j++)
                    printf("*");
                  printf("\n");
                }
                for (i=n2/2+1;i>=1;i--){ //下半
                  for (j=1;j<=n2/2+1-i;j++)
                    printf("#");
                  for (j=1;j<=i;j++)
                    printf("*");
                  printf("\n");
                }
              }
              else{ //n1=3: diamond
                for (i=1;i<=n2/2;i++){ //上半
                  for (j=1;j<=n2/2+1-i;j++)
                    printf("#");
                  for (j=1;j<=i*2-1;j++)
                    printf("*");
                  printf("\n");
                }
                for (i=n2/2+1;i>=1;i--){ //下半
                  for (j=1;j<=n2/2+1-i;j++)
                    printf("#");
                  for (j=1;j<=i*2-1;j++)
                    printf("*");
                  printf("\n");
                }
              }
            }

主程式 

            int main(){
              int choose; //選擇哪個function
              int n1, n2;  //要丟給function的數字
            
              while(1){ //無窮迴圈
                scanf("%d", &choose);
                if(choose == 0) //輸入0結束
                  break;
                else if(choose != 1 && choose != 2){
                  printf("Error\n");
                  break;
                }
                else if(choose == 1){
                  scanf("%d",&n1);
                  f1(n1);
                }
                else if(choose == 2){
                  scanf("%d\n%d",&n1,&n2);
                  f2(n1,n2);
                }
              }
              return 0;
            }
